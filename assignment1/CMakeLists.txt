ELSE(WIN32)
cmake_minimum_required(VERSION 2.4)
ENDIF(WIN32)

ENDIF(CMAKE_INSTALL_PREFIX_INITIALIZED_TO_DEFAULT)
ENDIF (NOT ADS_FOUND)
